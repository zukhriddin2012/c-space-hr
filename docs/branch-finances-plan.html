Database Schema

-- Core accounting tables

-- Chart of Accounts
CREATE TABLE accounts (
  id            UUID PRIMARY KEY,
  code          VARCHAR(10) NOT NULL UNIQUE,  -- "4100", "5200"
  name          VARCHAR(100) NOT NULL,        -- "Office Rental Income"
  name_uz       VARCHAR(100),
  name_ru       VARCHAR(100),
  type          VARCHAR(20) NOT NULL,         -- asset, liability, equity, revenue, expense
  parent_id     UUID REFERENCES accounts(id), -- for hierarchy
  is_active     BOOLEAN DEFAULT true,
  created_at    TIMESTAMP DEFAULT NOW()
);

-- Customers/Tenants (for receivables)
CREATE TABLE finance_customers (
  id            UUID PRIMARY KEY,
  branch_id     UUID NOT NULL REFERENCES branches(id),
  name          VARCHAR(200) NOT NULL,
  customer_type VARCHAR(20),                  -- company, individual
  telegram_id   VARCHAR(50),                  -- for reminders
  phone         VARCHAR(20),
  email         VARCHAR(100),
  notes         TEXT,
  created_at    TIMESTAMP DEFAULT NOW()
);

-- Journal Entries (double-entry core)
CREATE TABLE journal_entries (
  id            UUID PRIMARY KEY,
  branch_id     UUID NOT NULL REFERENCES branches(id),
  entry_date    DATE NOT NULL,
  reference     VARCHAR(50),                  -- invoice #, receipt #
  description   TEXT,
  status        VARCHAR(20) DEFAULT 'posted', -- draft, posted, void
  created_by    UUID REFERENCES employees(id),
  created_at    TIMESTAMP DEFAULT NOW()
);

-- Journal Entry Lines (debit/credit pairs)
CREATE TABLE journal_entry_lines (
  id            UUID PRIMARY KEY,
  journal_entry_id UUID NOT NULL REFERENCES journal_entries(id),
  account_id    UUID NOT NULL REFERENCES accounts(id),
  debit         DECIMAL(18,2) DEFAULT 0,
  credit        DECIMAL(18,2) DEFAULT 0,
  memo          VARCHAR(200)
);

-- Simplified Transaction View (what users see)
CREATE TABLE transactions (
  id              UUID PRIMARY KEY,
  branch_id       UUID NOT NULL REFERENCES branches(id),
  journal_entry_id UUID REFERENCES journal_entries(id),

  -- User-friendly fields
  transaction_type VARCHAR(20) NOT NULL,      -- revenue, expense, transfer
  transaction_date DATE NOT NULL,
  amount          DECIMAL(18,2) NOT NULL,

  -- Revenue specific
  service_type    VARCHAR(50),                -- office, flex, meeting, etc.
  customer_id     UUID REFERENCES finance_customers(id),
  customer_name   VARCHAR(200),               -- denormalized for display

  -- Expense specific
  expense_category VARCHAR(50),               -- goods, utility, maintenance
  vendor_name     VARCHAR(200),

  -- Payment info
  payment_method  VARCHAR(20),                -- cash, bank, payme, click, uzum, terminal
  payment_reference VARCHAR(100),             -- check #, transaction ID

  -- Tracking
  processed_by    UUID REFERENCES employees(id),
  approved_by     UUID REFERENCES employees(id), -- for expenses >500K
  approval_status VARCHAR(20) DEFAULT 'approved', -- pending, approved, rejected

  -- Import tracking
  imported_from   VARCHAR(50),                -- 'excel', 'manual'
  import_batch_id UUID,

  notes           TEXT,
  created_at      TIMESTAMP DEFAULT NOW()
);

-- Receivables (debt tracking)
CREATE TABLE receivables (
  id            UUID PRIMARY KEY,
  branch_id     UUID NOT NULL REFERENCES branches(id),
  customer_id   UUID NOT NULL REFERENCES finance_customers(id),
  invoice_number VARCHAR(50),
  invoice_date  DATE NOT NULL,
  due_date      DATE NOT NULL,
  amount        DECIMAL(18,2) NOT NULL,
  paid_amount   DECIMAL(18,2) DEFAULT 0,
  status        VARCHAR(20) DEFAULT 'open',   -- open, partial, paid, overdue, written_off
  last_reminder_at TIMESTAMP,
  reminder_count INTEGER DEFAULT 0,
  notes         TEXT,
  created_at    TIMESTAMP DEFAULT NOW()
);

-- Branch Cash Balances (for reconciliation)
CREATE TABLE branch_balances (
  id            UUID PRIMARY KEY,
  branch_id     UUID NOT NULL REFERENCES branches(id),
  balance_date  DATE NOT NULL,
  petty_cash    DECIMAL(18,2) DEFAULT 0,
  bank_balance  DECIMAL(18,2) DEFAULT 0,
  payme_balance DECIMAL(18,2) DEFAULT 0,
  click_balance DECIMAL(18,2) DEFAULT 0,
  uzum_balance  DECIMAL(18,2) DEFAULT 0,
  verified_by   UUID REFERENCES employees(id),
  notes         TEXT,
  UNIQUE(branch_id, balance_date)
);

-- Budgets
CREATE TABLE budgets (
  id            UUID PRIMARY KEY,
  branch_id     UUID NOT NULL REFERENCES branches(id),
  account_id    UUID NOT NULL REFERENCES accounts(id),
  year          INTEGER NOT NULL,
  month         INTEGER NOT NULL,             -- 1-12, or NULL for annual
  amount        DECIMAL(18,2) NOT NULL,
  notes         TEXT,
  UNIQUE(branch_id, account_id, year, month)
);

-- Import History
CREATE TABLE import_batches (
  id            UUID PRIMARY KEY,
  branch_id     UUID REFERENCES branches(id),
  file_name     VARCHAR(200),
  file_type     VARCHAR(20),                  -- revenue, cost, sales
  row_count     INTEGER,
  success_count INTEGER,
  error_count   INTEGER,
  errors        JSONB,                        -- detailed error log
  imported_by   UUID REFERENCES employees(id),
  created_at    TIMESTAMP DEFAULT NOW()
);

-- Indexes for performance
CREATE INDEX idx_transactions_branch_date ON transactions(branch_id, transaction_date);
CREATE INDEX idx_transactions_type ON transactions(transaction_type);
CREATE INDEX idx_journal_entries_branch_date ON journal_entries(branch_id, entry_date);
CREATE INDEX idx_receivables_customer ON receivables(customer_id);
CREATE INDEX idx_receivables_status ON receivables(status);
